104  break;

162 void brainfuck_parser_parse_string_partial(const char *string,
163  struct BrainfuckParserContext *ctx, int *error)
164 {
165  char ch;
166  int error_holder;
167  int delta;
168  int offset = 0;
169  unsigned int n;
170  int index;
171  size_t size;
172  struct BrainfuckInstruction *instruction;
173  struct BrainfuckInstruction *tmp;
174 
175  assert(ctx);
176 
177  error = error ? error : &error_holder;
178  *error = BRAINFUCK_EOK;
179 
180  /* Initialize the context if it hasn't been initialized yet */
181  if (!ctx->offsets.array && brainfuck_parser_context_init(ctx) != BRAINFUCK_EOK)
182  goto error_nomem;
183 
184  /* Initialize script if it hasn't been initialized yet */
185  if (!ctx->tail) {
186  ctx->tail = brainfuck_script_alloc();
187  if (!ctx->tail || brainfuck_script_init(ctx->tail, NULL, BRAINFUCK_ARRAY_SIZE) != BRAINFUCK_EOK)
188  goto error_nomem;
189  ctx->head = ctx->tail;
190  }
191  index = ctx->tail->index;
192  size = ctx->tail->size;
193  instruction = &ctx->tail->array[index];
194  offset = ctx->offsets.current;
195  while((ch = *string++)) {
196  switch(ch) {
197  case '+':
198  case '-':
199  delta = 0;
200  string--;
201  while ((ch = *string++) == '+' || ch == '-')
202  delta += (ch == '+' ? 1 : -1);
203  string--;
204  if (delta == 0) {
205  continue;
206  } else if (delta > 0) {
207  instruction->type = INC;
208  instruction->argument.offset = delta;
209  } else {
210  instruction->type = DEC;
211  instruction->argument.offset = -delta;
212  }
213  break;
214  case '>':
215  case '<':
216  delta = 0;
217  string--;
218  while ((ch = *string++) == '>' || ch == '<')
219  delta += (ch == '>' ? 1 : -1);
220  string--;
221  if (delta == 0) {
222  continue;
223  } else if (delta > 0) {
224  instruction->type = MOVR;
225  instruction->argument.offset = delta;
226  } else {
227  instruction->type = MOVL;
228  instruction->argument.offset = -delta;
229  }
230  break;
231  case ',':
232  n = 0;
233  string--;
234  while ((ch = *string++) == ',' && n++);
235  instruction->type = IN;
236  instruction->argument.n = n;
237  break;
238  case '.':
239  n = 0;
240  string--;
241  while ((ch = *string++) == '.' && n++);
242  instruction->type = OUT;
243  instruction->argument.n = n;
244  break;
245  case '[':
246  if ((size_t) ctx->offsets.index >= ctx->offsets.size)
247  goto error_nomem;
248  instruction->type = JZ;
249  instruction->argument.offset = offset;
250  ctx->offsets.array[ctx->offsets.index++] = instruction;
251  offset = 0;
252  break;
253  case ']':
254  if (ctx->offsets.index <= 0)
255  goto error_syntax;
256  instruction->type = JMP;
257  instruction->argument.offset = -offset;
258  tmp = ctx->offsets.array[--ctx->offsets.index];
259  offset += tmp->argument.offset;
260  tmp->argument.offset = -instruction->argument.offset + 1;
261  break;
262  default:
263  continue;
264  }
265  instruction += (index++, offset++, 1);
266  if ((size_t) index < size)
267  continue;
268  ctx->tail->next = brainfuck_script_alloc();
269  if (!ctx->tail->next ||
270  brainfuck_script_init(ctx->tail->next, ctx->tail, BRAINFUCK_ARRAY_SIZE) != BRAINFUCK_EOK)
271  goto error_nomem;
272  ctx->tail->index = index;
273  ctx->tail = ctx->tail->next;
274  index = ctx->tail->index;
275  size = ctx->tail->size;
276  instruction = &ctx->tail->array[index];
277  }
278  ctx->tail->index = index;
279  ctx->offsets.current = offset;
280  return;
281 
282  error_nomem:
283  *error = BRAINFUCK_ENOMEM;
284  return;
285  error_syntax:
286  *error = BRAINFUCK_ESYNTAX;
287  return;
288 }

291  int *error)

318 struct BrainfuckScript * brainfuck_parser_parse_file(FILE *file, int *error)
319 {
320  int error_holder = BRAINFUCK_EOK;
321  char buffer[BRAINFUCK_BUFFER_SIZE] = {0};
322  struct BrainfuckParserContext ctx = {NULL, NULL, {0, 0, 0, NULL}};
323 
324  error = error ? error : &error_holder;
325  *error = BRAINFUCK_EOK;
326 
327  assert(file);
328  while (!feof(file)) {
329  memset(&buffer, 0, sizeof(buffer));
330  fread(&buffer, sizeof(buffer), 1, file);
331  if (ferror(file))
332  continue;
333  brainfuck_parser_parse_string_partial(buffer, &ctx, error);
334  if (*error != BRAINFUCK_EOK)
335  goto error;
336  }
337 
338  if (!brainfuck_parser_validate(&ctx)) {
339  *error = BRAINFUCK_ESYNTAX;
340  goto error;
341  }
342  free(ctx.offsets.array);
343  return ctx.head;
344 
345  error:
346  free(ctx.offsets.array);
347  if (ctx.head)
348  brainfuck_script_dealloc(ctx.head);
349  return NULL;
350 }

352 static struct BrainfuckInstruction * next(const struct BrainfuckScript *script, int index) {
353  assert(script);
354 
355  while (index < 0 || index >= script->index) {
356  if (index >= script->index) {
357  index -= script->index;
358  script = script->next;
359  if (!script)
360  return NULL;
361  } else if (index < 0) {
362  script = script->previous;
363  if (!script)
364  return NULL;
365  index += script->index;
366  }
367  }
368  return script->array + index;
369 }

371 int brainfuck_pass_clear(const struct BrainfuckScript *script)
372 {
373  struct BrainfuckInstruction *instruction;
374  struct BrainfuckInstruction *a;
375  struct BrainfuckInstruction *b;
376  int i = 0;
377  while((instruction = next(script, i++))) {
378  if (instruction->type != JZ)
379  continue;
380  a = next(script, i++);
381  if (!a || !(a->type == INC ||a->type == DEC))
382  continue;
383  b = next(script, i++);
384  if (!b || !(b->type == JMP))
385  continue;
386  instruction->type = CLR;
387  a->type = NOP;
388  b->type = NOP;
389  }
390  return BRAINFUCK_EOK;
391 }

404 int brainfuck_engine_run(const struct BrainfuckScript *script,
405  struct BrainfuckEngineContext *ctx)
406 {
407  unsigned int i;
408  register int j;
409  register unsigned int k;
410  int size;
411  struct BrainfuckInstruction *instructions;
412  register unsigned char *memory = ctx->memory;
413  register int index = 0;
414  struct BrainfuckInstruction instruction;
415  union BrainfuckArgument argument;
416 
417 
418  assert(script);
419  assert(ctx);
420 
421  while (script) {
422  size = script->index;
423  instructions = script->array;
424  for (i = 0; i < (unsigned int) size; i++) {
425  instruction = instructions[i];
426  argument = instruction.argument;
427  switch(instruction.type) {
428  case INC:
429  /*printf("inc %p %i => 0x%08x\n", memory, instruction.argument.offset, *memory + instruction.argument.offset);*/
430  memory[index] += argument.offset;
431  break;
432  case DEC:
433  /*printf("dec %p %i => 0x%08x\n", memory, instruction.argument.offset, *memory + instruction.argument.offset);*/
434  memory[index] -= argument.offset;
435  break;
436  case MOVL:
437  /*printf("movl %p %i => %p\n", memory, instruction.argument.offset, memory + instruction.argument.offset);*/
438  index -= argument.offset;
439  break;
440  case MOVR:
441  /*printf("mov %p %i => %p\n", memory, instruction.argument.offset, memory + instruction.argument.offset);*/
442  index += argument.offset;
443  break;
444  case CLR:
445  /*printf("clr\n");*/
446  memory[index] = 0;
447  break;
448  case IN:
449  for (k = 0; k < argument.n; k++)
450  memory[index] = ctx->read();
451  /*printf("in %p => 0x%08x\n", memory, *memory);*/
452  break;
453  case OUT:
454  /*printf("out %p => 0x%08x\n", memory, *memory);*/
455  for (k = 0; k < argument.n; k++)
456  ctx->write(memory[index]);
457  break;
458  case JZ:
459  /*printf("jz %i\n", instruction.argument.offset);*/
460  if (memory[index])
461  break;
462  /*Fallthrough */
463  case JMP:
464  /*printf("jmp %i\n", instruction.argument.offset);*/
465  j = i + argument.offset - 1;
466  while (j < 0 || j >= size) {
467  if (j < 0) {
468  script = script->previous;
469  size = script->index;
470  j += size;
471  } else {
472  j -= size;
473  script = script->next;
474  size = script->index;
475  }
476  instructions = script->array;
477  }
478  i = j;
479  break;
480  case NOP:
481  break;
482  }
483  }
484  script = script->next;
485  }
486  return BRAINFUCK_EOK;
487 }
