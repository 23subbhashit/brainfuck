51  * The abstract syntax tree that should be utilized by a proper implementation.

56 /**
57  * This structure represents a single operation that could be executed. It
58  * contains the opcode of the operation and the arguments.
59  */
60 struct BrainfuckInstruction {
61  /**
62  * This enum defines the instruction set that should be implemented by a
63  * proper implementation of this specification.
64  */
65  enum BrainfuckType {
66  /* Data Manipulation */
67  INC, /**< Increase cell value */
68  DEC, /**< Decrease cell value */
69  MOVL, /**< Move left in the memory */
70  MOVR, /**< Move right in the memory */
71  CLR, /**< Clear cell value */
72  /* IO */
73  IN, /**< Read one byte from a stream and write to cell */
74  OUT, /**< Read one byte from cell and write to a stream */
75  /* Control Flow */
76  JMP, /**< Unconditional jump */
77  JZ, /**< Conditional jump: if zero */
78  /* Util */
79  NOP, /**< No operation */
80  } type;
81 
82  /**
83  * The argument of this instruction, whose type depends on the operation
84  * of the instruction.
85  */
86  union BrainfuckArgument {
87  /**
88  * The offset in cell value or instruction address.
89  */
90  int offset;
91 
92  /**
93  * The number of times an instruction should be executed.
94  */
95  unsigned int n;
96  } argument;
97 };
